UENUM(BlueprintType)
enum class EGOShipModuleState : uint8
{
	Active UMETA(DisplayName="Active"),
	Damaged UMETA(DisplayName="Damaged"),
	Critical UMETA(DisplayName="Critical"),
	Destroyed UMETA(DisplayName="Destroyed")
};

UCLASS(ClassGroup=(Custom), meta=(BlueprintSpawnableComponent))
class P4GALAXYONE_API UGOShipModuleComponent : public USceneComponent
{
	GENERATED_BODY()

public:
	UGOShipModuleComponent();

	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="GO|ShipModule")
	FName ModuleId = NAME_None;

	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="GO|ShipModule")
	EGOShipModuleState ModuleState = EGOShipModuleState::Active;

	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="GO|ShipModule|Damage")
	float ModuleIntegrity = 100.f;

	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="GO|ShipModule|Damage")
	float DamagedIntegrityThreshold = 70.f;

	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="GO|ShipModule|Damage")
	float CriticalIntegrityThreshold = 35.f;

	UFUNCTION(BlueprintCallable)
	bool IsModuleActive() const;

	UFUNCTION(BlueprintCallable)
	bool IsModuleDestroyed() const;

	UFUNCTION(BlueprintCallable)
	void SetModuleIntegrity(float NewIntegrity);

	UFUNCTION(BlueprintCallable)
	float ApplyModuleDamage(float DamageAmount);

	UFUNCTION(BlueprintCallable)
	float RepairModuleIntegrity(float RepairAmount);

	UFUNCTION(BlueprintCallable)
	virtual void UpdateModuleStateFromIntegrity();

protected:
	virtual void OnModuleStateChanged(EGOShipModuleState OldState, EGOShipModuleState NewState);
};

#include "ShipModules/GOShipModuleComponent.h"

UGOShipModuleComponent::UGOShipModuleComponent()
{
	PrimaryComponentTick.bCanEverTick = false;
	bVisualizeComponent = true;
}

bool UGOShipModuleComponent::IsModuleActive() const
{
	return ModuleState != EGOShipModuleState::Destroyed;
}

bool UGOShipModuleComponent::IsModuleDestroyed() const
{
	return ModuleState == EGOShipModuleState::Destroyed;
}

void UGOShipModuleComponent::SetModuleIntegrity(float NewIntegrity)
{
	ModuleIntegrity = FMath::Clamp(NewIntegrity, 0.f, 100.f);
	UpdateModuleStateFromIntegrity();
}

float UGOShipModuleComponent::ApplyModuleDamage(float DamageAmount)
{
	if (DamageAmount <= 0.f || IsModuleDestroyed())
	{
		return 0.f;
	}

	const float OldIntegrity = ModuleIntegrity;
	SetModuleIntegrity(ModuleIntegrity - DamageAmount);

	return OldIntegrity - ModuleIntegrity;
}

float UGOShipModuleComponent::RepairModuleIntegrity(float RepairAmount)
{
	if (RepairAmount <= 0.f)
	{
		return 0.f;
	}

	const float OldIntegrity = ModuleIntegrity;
	SetModuleIntegrity(ModuleIntegrity + RepairAmount);

	return ModuleIntegrity - OldIntegrity;
}

void UGOShipModuleComponent::UpdateModuleStateFromIntegrity()
{
	const EGOShipModuleState OldState = ModuleState;
	EGOShipModuleState NewState = EGOShipModuleState::Active;

	if (ModuleIntegrity <= 0.f)
	{
		NewState = EGOShipModuleState::Destroyed;
	}
	else if (ModuleIntegrity <= CriticalIntegrityThreshold)
	{
		NewState = EGOShipModuleState::Critical;
	}
	else if (ModuleIntegrity <= DamagedIntegrityThreshold)
	{
		NewState = EGOShipModuleState::Damaged;
	}

	ModuleState = NewState;

	if (OldState != ModuleState)
	{
		OnModuleStateChanged(OldState, ModuleState);
	}
}

Portfolio explanation text
------------------------------------------------------------

This component is the base layer of Galaxy One's modular ship damage system.
Each ship module owns an integrity value and automatically transitions between Active, Damaged, Critical, and Destroyed states based on configurable thresholds.
The component exposes Blueprint-callable damage and repair functions, while OnModuleStateChanged provides an extension point for specialized modules such as engines, power systems, weapons, fuel lines, or sensors.
